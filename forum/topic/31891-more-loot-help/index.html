        Posted January 29, 2015 
        _list= position player nearObjects ["House",50];
{
	_building = _x;
	_bPos = [_building] call BIS_fnc_buildingPositions;
	if (count _bPos > 0) then {
		{
			for "_moreLoot" from 1 to 20 do{
				_lootholder = createVehicle ["WeaponHolderSimulated", _x, [], 0, "CAN_COLLIDE"];
				_lootholder addWeaponCargoGlobal ["m14_epoch",1];
				_ammo = [] + getArray (configFile >> "cfgWeapons" >> "m14_epoch" >> "magazines");
				_mag = _ammo select (floor(random(count _ammo)));
				_lootholder addMagazineCargoGlobal [_mag,round(floor(random 3))];
				_lootholder setdir (random 360);
				_lootholder setPos [(_x select 0),(_x select 1),100];
			};
		} foreach _bPos;
	};
} forEach _list;

        just run arma with -mod=@Epoch , Launch the editor (any map) , place a player beside some buildings an run the script from the debug console
