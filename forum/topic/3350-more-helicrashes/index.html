        Not no all variables are present in the file, but you can add them yourself, i have all these in my config section:

        // DayZ Epoch config
spawnShoremode = 0; // Default = 1 (on shore)
dayz_paraSpawn = false;
spawnArea= 1500; // Default = 1500
MaxHeliCrashes= 7; // Default = 5
MaxVehicleLimit = 600; // Default = 50
MaxDynamicDebris = 100; // Default = 100
dayz_MapArea = 14000; // Default = 10000
dayz_maxLocalZombies = 30; // Default = 30
dayz_maxGlobalZombiesInit = 10;
dayz_maxGlobalZombiesIncrease = 5;
dayz_maxZeds = 400;
dayz_maxAnimals = 8; // Default: 8
dayz_tameDogs = true;
DynamicVehicleDamageLow = 0; // Default: 0
DynamicVehicleDamageHigh = 30; // Default: 100
dayz_sellDistance = 30;
dayz_zedsAttackVehicles = false;
DynamicVehicleFuelLow = 30;
DZE_DeathMsgGlobal = true;
DZE_teleport = [1000,2000,500,20000,20000];
DZE_BuildingLimit = 500;
DZE_AllowForceSave = true;

EpochEvents = [
["any","any","any","any",45,"scavenger4"],
["any","any","any","any",15,"ikea"],
["any","any","any","any",30,"crash_spawner"],
["any","any","any","any",35,"supply_drop"],
["any","any","any","any",00,"scavenger_hunt"]
];
dayz_fullMoonNights = true;

        I guess there are some missing in the wiki here: http://dayzepoch.com/wiki/index.php?title=Init.sqf

         

        But if you want take a look at the file "DayZ_Epoch_Server_1.0.2.5_Release\@DayZ_Epoch_Server\addons\dayz_server\system\server_monitor.sqf" around line 48:

        // Custom Configs
if(isnil "MaxVehicleLimit") then {
	MaxVehicleLimit = 50;
};
if(isnil "MaxHeliCrashes") then {
	MaxHeliCrashes = 5;
};
if(isnil "MaxDynamicDebris") then {
	MaxDynamicDebris = 100;
};
if(isnil "MaxAmmoBoxes") then {
	MaxAmmoBoxes = 3;
};
if(isnil "MaxMineVeins") then {
	MaxMineVeins = 50;
};
// Custon Configs End

        So there are other config variables not defined in the init.sqf by default. :)
