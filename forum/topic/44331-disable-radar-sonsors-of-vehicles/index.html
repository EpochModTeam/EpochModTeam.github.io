        If somebody is also looking for a script function to disable the Radar of Vehicles:
          

        	{
		_sensor = _x;
		if (_sensor isequaltype []) then {
			if (count _sensor > 0) then {
				_sensor = _x select 0;
			};
		};
		_vehicle enableVehicleSensor [_sensor,false];
	} foreach (listVehicleSensors _vehicle);
