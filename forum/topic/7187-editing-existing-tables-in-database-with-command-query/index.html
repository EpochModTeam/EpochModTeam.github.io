        You mean something like this?

        Â 

        UPDATE `traders_data` SET `buy`='[4,"ItemGoldBar10oz",1]' WHERE `item`='["30m_plot_kit",1]';
        UPDATE `traders_data` SET `sell`='[2,"ItemGoldBar10oz",1]' WHERE `item`='["30m_plot_kit",1]';
