        Ok, I might be able to help you out with script 2. The randomizing for the loot that I use is like this:

        x=floor(random 3);    //0-2


if (x == 0) then {
_crate3 = _this select 0;

clearWeaponCargoGlobal _crate3;
clearMagazineCargoGlobal _crate3;

_crate3 addWeaponCargoGlobal ["FN_FAL_ANPVS4", 2];


_crate3 addBackpackCargoGlobal ["DZ_Backpack_EP1", 2];

_crate3 addMagazineCargoGlobal ["20Rnd_762x51_FNFAL", 20];

 };
if (x == 1) then {
_crate3 = _this select 0;

clearWeaponCargoGlobal _crate3;
clearMagazineCargoGlobal _crate3;

_crate3 addWeaponCargoGlobal ["Sa58P_EP1", 2];

_crate3 addBackpackCargoGlobal ["DZ_GunBag_EP1", 1];

_crate3 addMagazineCargoGlobal ["20Rnd_762x51_FNFAL", 20];

 };

if (x == 2) then {
_crate3 = _this select 0;

clearWeaponCargoGlobal _crate3;
clearMagazineCargoGlobal _crate3;

_crate3 addWeaponCargoGlobal ["RPK_74", 2];

_crate3 addBackpackCargoGlobal ["DZ_British_ACU", 1];

_crate3 addMagazineCargoGlobal ["75Rnd_545x39_RPK", 5];

 };

        This would allow you to make multiple combinations. The amount of combinations can be changed very easily, read through the script and you should be able to figure it out.
