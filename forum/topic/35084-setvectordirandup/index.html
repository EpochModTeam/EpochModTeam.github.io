        Open this and press CTRL & F and search for attach: https://community.bistudio.com/wiki/setVectorDirAndUp

        _object = createVehicle [_classname, _location, [], 0, "CAN_COLLIDE"];
	//Build gizmo
	_objectHelper = "Sign_sphere10cm_EP1" createVehicle _location;
	_helperColor = "#(argb,8,8,3)color(0,0,0,0,ca)";
	_objectHelper setobjecttexture [0,_helperColor];
	_objectHelper attachTo [player,_offset];
	_object attachTo [_objectHelper,[0,0,0]];
	//_position = [_objectHelper] call FNC_GetPos;
	_position = getPos _objectHelper;
	_centerSign = objNull;
	if (_classname == "CinderWallDoorway_Preview_DZ" || _classname == "CinderWallSmallDoorway_Preview_DZ") then {
			//systemchat "Arrow is testing, i can show wrong direction!";
			_centerSign = "Sign_arrow_down_large_EP1" createVehicle _location;
			_centerSign attachTo [_object,[(_offset select 0),(_offset select 1),1]];
			[_centerSign,[-80,0,0]] call fnc_SetPitchBankYaw;
		};

        if(DZE_5) exitWith {
			_isOk = false;
			_position = [_object] call FNC_GetPos;

			_vector = [(vectorDir _object),(vectorUp _object)];
