        Here's my randomized loot for the box if someone wants it:

        (Tested and working)

        private ["_amount","_types"]

if (isServer) then {
	_vehicle_999999 = objNull;
		
	if (true) then {
		_this = createVehicle ["TKVehicleBox_EP1", [55.198521,1580.6722,4.7683716e-007], [], 0, "CAN_COLLIDE"];
		_vehicle_999999 = _this;
		_this setDir 20;
		_vehicle_999999 setVariable ["ObjectID","1",true];
                _vehicle_999999 setVariable ["permaLoot",true];
		_rarity = 0;

		clearWeaponCargoGlobal _this;
		clearMagazineCargoGlobal _this;

		_gearChances=floor(random 1);
		if (_gearChances == 0) then {
			_rarity = 1;
		} else {
			_rarity = 0;
		};
		
		if (_rarity == 1) then {
		_amountSuperRare = 1;
		_amountRare = 2;
		_amountCommon = 3;
		} else {
			_randomizeAll = 1;
		};
		
		if (_randomizeAll == 1) then {
		_amountSuperRare=floor(random 1);
		_this addWeaponCargoGlobal ["ItemToolbox",_amountSuperRare]
		_amountRare=floor(random 1);
		_this addWeaponCargoGlobal ["ItemCrowbar",_amountRare];
		_amountRare=floor(random 1);
		_this addWeaponCargoGlobal ["Colt1911",_amountRare];
		_amountRare=floor(random 3);
		_this addWeaponCargoGlobal ["Makarov",_amountRare];
		_amountRare=floor(random 1);
		_this addWeaponCargoGlobal ["Winchester1866",_amountRare];
		_amountCommon=floor(random 2);
		_this addMagazineCargoGlobal ["8Rnd_9x18_Makarov",_amountCommon];
		_amountCommon=floor(random 4);
		_this addMagazineCargoGlobal ["7Rnd_45ACP_1911",_amountCommon];
		_amountRare=floor(random 3);
		_this addMagazineCargoGlobal ["15Rnd_W1866_Slug",_amountRare];
		_amountCommon=floor(random 2);
		_this addMagazineCargoGlobal ["SmokeShell",_amountCommon];
		_amountCommon=floor(random 1);
		_this addWeaponCargoGlobal ["ItemCompass",_amountCommon];
		_amountRare=floor(random 1);
		_this addWeaponCargoGlobal ["ItemGPS",_amountRare];
		} else {
		_this addWeaponCargoGlobal ["ItemToolbox",_amountSuperRare]
		_this addWeaponCargoGlobal ["ItemCrowbar",_amountRare];
		_this addWeaponCargoGlobal ["Colt1911",_amountRare];
		_this addWeaponCargoGlobal ["Makarov",_amountRare];
		_this addWeaponCargoGlobal ["Winchester1866",_amountRare];
		_this addMagazineCargoGlobal ["8Rnd_9x18_Makarov",_amountCommon];
		_this addMagazineCargoGlobal ["7Rnd_45ACP_1911",_amountCommon];
		_this addMagazineCargoGlobal ["15Rnd_W1866_Slug",_amountRare];
		_this addMagazineCargoGlobal ["SmokeShell",_amountCommon];
		_this addWeaponCargoGlobal ["ItemCompass",_amountCommon];
		_this addWeaponCargoGlobal ["ItemGPS",_amountRare];
	};
};
