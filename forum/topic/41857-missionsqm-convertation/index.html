        Hi there, anyone can give an advice how to convert mission.sqm from pre-3DEN format to a newer (you can compare diffreneces from 0.3.7 and 0.3.8 builds of Epoch).

        ...
class Mission {

class Intel{};
class Groups{};
class Vehicles{};
class Markers{};
};
...

        Newer: 

        class Mission {
    class Entities {
        Item0 {
            type = "Group";
            ...
        };
        Item2 {
            type = "Vehicle";
            ...
        };
        Item2 {
            type = "Marker";
            ...
        };
    };
};
