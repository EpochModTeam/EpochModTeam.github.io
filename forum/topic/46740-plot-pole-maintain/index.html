        UPDATE `object_data` 
		SET `Damage`=0.1 
	WHERE `ObjectUID` <> 0 
		AND `CharacterID` <> 0 
		AND `Datestamp` < DATE_SUB(CURRENT_TIMESTAMP, INTERVAL 1 DAY) 
		AND ( (`Inventory` IS NULL) OR (`Inventory` = '[]') )
		AND ( (`Classname` NOT LIKE '%Vault%') AND (`Classname` NOT LIKE '%Lockbox%') )
