        There is a cleanup feature in the server config (epochconfig.hpp).

        expiresBuilding = "2592000";  // expiration date in seconds for buildings
expiresPlayer = "2592000";  // expiration date in seconds for players
expiresVehicle = "2592000";  // expiration date in seconds for vehicles
