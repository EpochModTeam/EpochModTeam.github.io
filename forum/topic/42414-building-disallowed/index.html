        _restrictedLocations = nearestLocations [player, ["NameCityCapital"], 300];
          if !(_restrictedLocations isEqualTo []) then {
          _buildingAllowed = false;
          ["Building Disallowed: Area Blocked", 5] call Epoch_message;
          };
