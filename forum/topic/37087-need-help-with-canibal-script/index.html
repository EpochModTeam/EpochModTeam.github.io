        There's no point checking for isHarvested twice, plus the error you got is because you never defined what isHarvested does in fn_selfActions. Still not required...
        You'll need to choose where to place it. If placing in selfActions, then action will not show up if body is gutted (and lead to confusion). If placing inside script then action will show up, but nothing will happen if already gutted.

        Here try this.

        //####    Gut fools ####
//ray: do as less checks here as possible since this file is already bad on performance, extra check inside external script could be good for RP messages
if (!_isAlive && !_isZombie && !_isAnimal && _isMan && _canDo) then {
	if (s_player_butcher_human < 0) then {
		s_player_butcher_human = player addAction [format["Gut Fool"], "custom\canibalize\gather_meat_human.sqf",(typeOf cursorTarget), 0, false, true, "", ""];
	};
} else {
	player removeAction s_player_butcher_human;
	s_player_butcher_human = -1;
};

//##############################  

//gather_meat_human.sqf
private["_hasKnife","_qty","_body","_text","_string","_timer","_dayzHumanity"];
_body = _this select 3;
_hasKnife =    "ItemKnife" in items player;
_hasHarvested = _body getVariable["meatHarvested",false];
 
player removeAction s_player_butcher;
s_player_butcher = -1;

if  (_hasHarvested) exitWith {cutText ["Body is already gutted.", "PLAIN DOWN"];};
if !(_hasKnife) exitWith {cutText ["You are missing knife to gut this body.", "PLAIN DOWN"];};

_rnd = random(100);

player playActionNow "Medic";
[player,"gut",0,false] call dayz_zombieSpeak;
_body setVariable["meatHarvested",true,true];

if (_rnd > 90) then {
	r_player_inpain = true;
	player setVariable["USEC_inPain",true,true];
	r_player_blood = r_player_blood - 500;
	cutText ["You have accidentally cut your finger.", "PLAIN DOWN"];
};

_qty = (random 3);

[player,50,true,(getPosATL player)] spawn player_alertZombies;

_dayzHumanity = [player,-100];
_dayzHumanity spawn player_humanityChange;


for "_x" from 1 to _qty do {
	_body addMagazine "FoodSteakRaw";
	_result = [player,"FoodSteakRaw"] call BIS_fnc_invAdd; //you are giving steak to both dead body above and player at the same time, why?
	if (_result) then {

	} else {
		cutText ["You didn't have enough room to store the meat.", "PLAIN DOWN"];
		_x = _qty;
	}
};

sleep 8;
