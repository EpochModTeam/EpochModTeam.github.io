        This may work for you. I can't test it on Epoch with sidechat but confirmed working on vanilla dayz with sidechat.

         

        put this in your ArmA2OAProfile

        class Difficulties
{
    class Recruit
    {
        class Flags
        {
            Armor=1;
            FriendlyTag=0;
            EnemyTag=0;
            HUD=1;
            HUDPerm=1;
            HUDWp=1;
            HUDWpPerm=1;

            AutoAim=1;
            AutoGuideAT=1;
            3rdPersonView=1;
            ClockIndicator=1;
            Map=1;
            Tracers=0;
            AutoSpot=1;
            UltraAI=0;
            UnlimitedSaves=1;
            DeathMessages=1;
            NetStats=1;
            VonID=1;
        };
        skillFriendly=1;
        precisionFriendly=1;
        skillEnemy=0.55000001;
        precisionEnemy=0.30000001;
    };
    class Regular
    {
        class Flags
        {
            3rdPersonView=1;

            hudGroupInfo=1;
            hudPerm=1;
            hudWp=1;
            hudWpPerm=1;
            map=1;
            netStats=1;
            tracers=0;
            ultraAI=0;
            unlimitedSaves=1;

        precisionEnemy=0.75;

        class Flags

            3rdPersonView=0;
            armor=0;
