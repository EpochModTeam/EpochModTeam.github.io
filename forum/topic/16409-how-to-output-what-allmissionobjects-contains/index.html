        Just do something like...

        _cars = 0;
_boat = 0;
_loot = 0;
_dead = 0;
_other = 0;

{
    if (_x isKindOf "WeaponHolder") then
    {
        _loot = _loot + 1;
    }
    else
    {
        if (_x isKindOf "Car") then
        {
            _cars = _cars +1;
        }
        else
        {
            // More "if" stuff goes here.
        };
    };
} forEach allMissionObjects;

diag_log format["LOOT: %1 CARS: %2 BOATS: %3 DEAD: %4 TOTAL: %5",_loot,_cars,_boat,_dead,_other];

        Or something.
