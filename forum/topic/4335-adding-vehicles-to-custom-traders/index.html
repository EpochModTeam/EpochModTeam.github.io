        This is the entirety of my script for one spawn spot for a custom trader...works nice.  Uses the yellow outlined Civilian heli spot.  :

         

        if (isServer) then {
         
        _vehicle_98 = objNull;
        if (true) then
        {
          _this = createVehicle ["HeliHEmpty", [13202.5,10570.7,0.002], [], 0, "CAN_COLLIDE"];
          _vehicle_98 = _this;
          _this setDir 91;
          _this setPos [13202.5,10570.7,0.002];
        };
         
        _vehicle_99 = objNull;
        if (true) then
        {
          _this = createVehicle ["HeliHCivil", [13202.5,10570.7,0.002], [], 0, "CAN_COLLIDE"];
          _vehicle_99 = _this;
          _this setDir 91;
          _this setPos [13202.5,10570.7,0.002];
        };
         
        };
         
         
         
        This spot is for a custom trader I have on top of a mountain, Taviana map.
        The setDir seems to spawn vehicles facing in the direction you specify (360 degrees)
