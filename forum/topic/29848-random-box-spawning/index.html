        private ['_position','_boxCount','_crate'];
_boxCount = N; // How many to spawn?

if (isServer) then {
    for '_i' from 1 to _bocCount do { 
	_position = [(getMarkerPos 'center'),0,15000,10,0,2,0] call BIS_fnc_findSafePos; // More info on this function here.
	_crate = createVehicle ['BOX_CLASS_NAME',_position,[],0,'CAN_COLLIDE'];
	_crate setVariable ['permaLoot',true,true];
	clearWeaponCargoGlobal _crate;
	clearMagazineCargoGlobal _crate;
	_crate spawn // Allows us to sleep 120 seconds before deleting without halting progress on creating more boxes.
	{
	    uiSleep 120;
	    deleteVehicle _this;
	};
    };
};

        This'll spawn in N amount of crates for you.

        You can learn more on how to properly call BIS_fnc_findSafePos here.
