        dbgTimeplU = time;

        };
        };

        if (swDebugLS) then {
        dbgloopTimeplU = dbgloopTimeplU + (time - dbgTimeplU);
        dbgTurnsplU = dbgTurnsplU + 1;
        };
        }forEach playableUnits;

        dbgloopTime = dbgloopTime + dbgloopTimeplU;
        dbgloopTimeplU = 0;

        dbgTurns = 0;
        dbgTurnsplU = 0;
        dbgloopTime = 0;
        };
        };

        if (!_playersalive) then {
        sleep 2;
        };

        };
